444template <typename T>
445std::vector<std::string> Menu<T>::getParams(std::string command, std::string operation, int paramCount)
446{
447 std::vector<std::string> params;
448 std::istringstream ss(command);
449 std::string t;
450 try {
451 std::getline(ss, t, ' '); // gets the parameters as pure strings
452 }
453 catch (std::exception) {
454 throw std::invalid_argument("Invalid command format. Check help for proper format");
455 }
456
457 char del = '|';
458
459 while (std::getline(ss, t, del))

461 params.push_back(t);
